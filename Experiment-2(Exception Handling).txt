#include    <iostream>
using namehpace std;
int main(){
    int n,int arr[n], int fliparr[n];
    cin>>n;
    cout<<n;
    int prev_ones=0;
    for( int i=0;i<n;i++){
        cin>>arr[i];
    }
    for( int i=0;i<n;i++){
        cout<<arr[i];
    }
    for( i=0;i<n;i++){
        if(arr[i]==1){
            prev_ones++;
            fliparr[i]=-1;
        }
        else{
            fliparr[i]=1;

        }
    }
    int max_so_far=0;
    int current_max = 0;

        for (int x : fliparr) {
            current_max += x;
            if (current_max > max_so_far) {
                max_so_far = current_max;
            }
            if (current_max < 0) {
                current_max = 0;
            }
        }
        cout << prev_ones + max_so_far << endl;
    return 0;
}
